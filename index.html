${exp.title}
${exp.company} | ${exp.start} - ${exp.end}
${exp.desc}
`; }); let skillsHTML = ''; data.skills.forEach(skill => { skillsHTML += `
${skill.name}
`; }); return `
${initials}
${data.fullName}
${data.jobTitle}
معلومات الاتصال
📧 ${data.email}
📱 ${data.phone}
${data.linkedin ? `🌐 ${data.linkedin}
` : ''} 📍 ${data.city}
${skillsHTML ? `
المهارات
${skillsHTML}
` : ''}
${data.summary ? `
نبذة شخصية
${data.summary}
` : ''} ${experiencesHTML ? `
الخبرات العملية
${experiencesHTML} ` : ''} ${data.degree ? `
المؤهلات العلمية
${data.degree}
${data.university} - ${data.gradYear}
${data.gpa ? `
التقدير: ${data.gpa}
` : ''}
` : ''}
`; } // طباعة السيرة الذاتية function printCV() { try { const formData = collectFormData(); const cvHTML = generateFullCV(formData); // فتح نافذة طباعة جديدة const printWindow = window.open('', '_blank'); printWindow.document.write(` السيرة الذاتية - ${formData.fullName} ${cvHTML} `); printWindow.document.close(); // انتظار تحميل المحتوى ثم الطباعة setTimeout(() => { printWindow.print(); printWindow.close(); }, 500); } catch (error) { console.error('Error printing CV:', error); alert('حدث خطأ أثناء الطباعة. جرب مرة أخرى.'); } } منشئ السيرة الذاتية الاحترافي
